import java.util.Scanner;

public class sistemAescola {

        public static void main(String[] args) {

            Scanner sc = new Scanner(System.in);

            //Entradas de dados
            System.out.println("Digite a nota 1");
            double nota1 = sc.nextDouble();

            System.out.println("Digite a nota 2");
            double nota2 = sc.nextDouble();

            System.out.println("Digite a nota 3");
            double nota3 = sc.nextDouble();

            System.out.println("Digite a nota 4");
            double nota4 = sc.nextDouble();

            System.out.println("Digite a nota 5");
            double nota5 = sc.nextDouble();

            System.out.println("Digite a nota 6");
            double nota6 = sc.nextDouble();

            System.out.println("Digite a nota 7");
            double nota7 = sc.nextDouble();

            System.out.println("Digite a nota 8");
            double nota8 = sc.nextDouble();

            // Processamento
            double notaBimestre1 = (nota1 + nota2) / 2;
            double notaBimestre2 = (nota3 + nota4) / 2;
            double notaBimestre3 = (nota5 + nota6) / 2;
            double notaBimestre4 = (nota7 + nota8) / 2;
            double notaSemestre1 = (notaBimestre1 + notaBimestre2) / 2;

            double notaSemestre2 = (notaBimestre3 + notaBimestre4) / 2;

            double notaFinal = (notaSemestre1 + notaSemestre2) / 2;


            // Saídas
            System.out.println("-------------------------------");
            System.out.println("A média do bimestre 1 é " + notaBimestre1);
            System.out.println("A média do bimestre 2 é " + notaBimestre2);
            System.out.println("A média do semestre 1 é " + notaSemestre1);
            System.out.println("A média do bimestre 3 é " + notaBimestre3);
            System.out.println("A média do bimestre 4 é " + notaBimestre4);
            System.out.println("A média do semestre 2 é " + notaSemestre2);
            System.out.println("A média final é " + notaFinal);
        }
    }
