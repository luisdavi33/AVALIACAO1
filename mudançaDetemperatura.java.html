import java.util.Scanner;

public class mudançaDetemperatura {
    public static void main(String[] args) {

        Scanner sc = new Scanner(System.in);

        //Entradas de dados
        System.out.println("Informe a temperatura em graus celsius ");
        double temperaturaCelsius = sc.nextDouble();

        double Fahrenheit = (temperaturaCelsius * 9/5) + 32;
        double Kelvin = temperaturaCelsius + 273.15;

        System.out.println("A temperatura em Celsius é  " + temperaturaCelsius);
        System.out.println("A temperatura em Fahrenheit é  " + Fahrenheit);
        System.out.println("A temperatura em Kelvin é  " + Kelvin);

    }
}
